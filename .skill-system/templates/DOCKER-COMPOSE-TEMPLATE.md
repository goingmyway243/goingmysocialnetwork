# Docker Compose Template\n\nuse this as a starting point for local development docker-compose file.\n\n```yaml\nversion: '3.8'\n\nservices:\n  # PostgreSQL Database\n  postgres:\n    image: postgres:13-alpine\n    container_name: goingmy-postgres\n    environment:\n      POSTGRES_DB: ${DB_NAME:-goingmy}\n      POSTGRES_USER: ${DB_USER:-postgres}\n      POSTGRES_PASSWORD: ${DB_PASSWORD:-postgres}\n    ports:\n      - \"${DB_PORT:-5432}:5432\"\n    volumes:\n      - postgres_data:/var/lib/postgresql/data\n    healthcheck:\n      test: [\"CMD-SHELL\", \"pg_isready -U ${DB_USER:-postgres}\"]\n      interval: 10s\n      timeout: 5s\n      retries: 5\n    networks:\n      - goingmy-network\n\n  # Redis Cache\n  redis:\n    image: redis:7-alpine\n    container_name: goingmy-redis\n    ports:\n      - \"${REDIS_PORT:-6379}:6379\"\n    healthcheck:\n      test: [\"CMD\", \"redis-cli\", \"ping\"]\n      interval: 10s\n      timeout: 5s\n      retries: 5\n    networks:\n      - goingmy-network\n\n  # RabbitMQ Message Queue\n  rabbitmq:\n    image: rabbitmq:3.11-management-alpine\n    container_name: goingmy-rabbitmq\n    ports:\n      - \"${RABBITMQ_PORT:-5672}:5672\"\n      - \"${RABBITMQ_MGMT_PORT:-15672}:15672\"\n    environment:\n      RABBITMQ_DEFAULT_USER: ${RABBITMQ_USER:-guest}\n      RABBITMQ_DEFAULT_PASS: ${RABBITMQ_PASS:-guest}\n    healthcheck:\n      test: [\"CMD\", \"rabbitmq-diagnostics\", \"-q\", \"ping\"]\n      interval: 10s\n      timeout: 5s\n      retries: 5\n    networks:\n      - goingmy-network\n\nnets:\n  goingmy-network:\n    driver: bridge\n\nvolumes:\n  postgres_data:\n```\n\n## Usage\n\n```bash\n# Start all services\ndocker-compose up -d\n\n# Check status\ndocker-compose ps\n\n# View logs\ndocker-compose logs -f\n\n# Stop all\ndocker-compose down\n\n# Remove volumes (reset data)\ndocker-compose down -v\n```\n\n## Environment Variables\n\nCreate `.env` file:\n```\nDB_NAME=goingmy\nDB_USER=postgres\nDB_PASSWORD=your_secure_password\nDB_PORT=5432\nREDIS_PORT=6379\nRABBITMQ_PORT=5672\nRABBITMQ_MGMT_PORT=15672\nRABBITMQ_USER=guest\nRABBITMQ_PASS=guest\n```\n"
